# ================================
# SECTION 10: Pipes
# ================================
cat /etc/passwd | grep root
ls -la | grep txt
echo hello world | wc -l
cat /etc/passwd | head -5 | tail -2
echo test line | cat | cat
ls thisfiledoesnotexist | grep a | more
cat /etc/passwd | grep root | head -3
echo multiple lines | tr ' ' '\n' | sort
date | cat | wc -c

# ================================
# SECTION 11: Pipes & Redirections
# ================================
cat /etc/passwd | grep root > root_users.txt
ls -la | grep txt >> file_list.txt 
echo hello | tee greeting.txt | cat
cat /etc/passwd | head -5 | tee top_users.txt | wc -l
echo test | tee count.txt | wc -l
ls thisfiledoesnotexist | grep a > error_output.txt
cat < /etc/passwd | grep root | head -2
echo data | wc -c
rm -f root_users.txt file_list.txt greeting.txt top_users.txt count.txt error_output.txt

# ================================
# SECTION 13: Logical Operators (Bonus)
# ================================
echo hello && echo world
echo hello || echo world
/bin/ls /etc/passwd && echo "file exists"
/bin/ls /nonexistent || echo "file not found"
echo test && echo success || echo failure
(echo first && echo second) || echo third
/bin/true && echo "true worked" || echo "true failed"
/bin/false || echo "false worked" && echo "continuing"

(false) || echo "subshell failed"
(true) && echo "subshell success"
